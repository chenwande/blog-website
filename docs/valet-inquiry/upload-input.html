    # Input 组件源码

    import { removeEmoji } from '@/utils';
import { CloseCircleFilled } from '@ant-design/icons';
import classNames from 'classnames';
import React, { forwardRef, ReactNode, useState } from 'react';
import styles from './index.module.scss';

interface IProps extends Omit<React.InputHTMLAttributes<HTMLInputElement>,'value' | 'onChange'> {
  value?: string;
  onChange: (value: string) => void;
  onClear?: () => void;
  isShowFocusStyle?: boolean;
  isShowClear?: boolean;
  isShowCount?: boolean;
  isShowEmptyError?: boolean;
  isFilterEmoji?: boolean;
  leftExtra?: ReactNode;
  rightExtra?: ReactNode;
}
const Input = forwardRef<HTMLInputElement, IProps>(
  (
    {
      value = '',
      onChange,
      onClear,
      onFocus,
      onBlur,
      maxLength,
      isShowClear = true,
      isShowCount = false,
      isShowEmptyError = false,
      isFilterEmoji = true,
      leftExtra,
      rightExtra,
      ...restProps
    },
    ref
  ) => {
    const handleClear = () => {
      onChange('');
      onClear?.();
    };

    const [isFocus, setIsFocus] = useState(false);
    const handleFocus = (args: any) => {
      setIsFocus(true);
      onFocus?.(args);
    };
    const handleBlur = (e: any) => {
      setIsFocus(false);
      onBlur?.(e);
    };
    const handleChange = (e: React.ChangeEvent<HTMLInputElement>) =>
      onChange(isFilterEmoji ? removeEmoji(e.target.value) : e.target.value);

    const countColor = value.length === maxLength ? '#00C142' : '#E51E1E';
    const showEmptyError = isShowEmptyError && !value && !isFocus;
    return (
      <div
        className={classNames([
          styles.container,
          isFocus && styles.focusBorder,
          showEmptyError && styles.emptyErrorContainer,
        ])}
      >
        {leftExtra}
        <input
          ref={ref}
          className={styles.input}
          maxLength={maxLength}
          type="text"
          value={value}
          onChange={handleChange}
          onFocus={handleFocus}
          onBlur={handleBlur}
          spellCheck={false}
          {...restProps}
        />
        {isShowClear && value && isFocus && (
          <span className={styles.closeIconBox} onMouseDown={handleClear}>
            <CloseCircleFilled className={styles.clear} />
          </span>
        )}
        {isShowCount && (
          <div className={styles.countContainer}>
            <span style={value ? { color: countColor } : undefined}>
              {value.length}
            </span>
            <span>/{maxLength}</span>
          </div>
        )}
        {rightExtra}
      </div>
    );
  }
);

export default Input;

    # UploadInput组件封装

    # 知识点

    • 拖拽监听

      通过监听dragover、dragleave、drop这个三个事件实现拖拽，isDragover状态可用控制展示拖拽悬停到拖拽区域的样式，drop事件中通过e.dataTransfer?.files拿到File对象从而调用上传

      const [isDragover, setIsDragover] = useState();
const handleDragover = useMemoizedFn((e: DragEvent) => {
  e.preventDefault();
  if (!isDragover) {
    setIsDragover(true);
  }
});
const handleDragleave = useMemoizedFn(() => {
  setIsDragover(false);
});
const handleDrop = useMemoizedFn(async (e: DragEvent) => {
  e.preventDefault();
  e.stopPropagation();
  setIsDragover(false);
  const files = e.dataTransfer?.files && Array.from(e.dataTransfer?.files);
  upload(files);
});
useEffect(() => {
  dropViewRef.current?.addEventListener('dragover', handleDragover);
  dropViewRef.current?.addEventListener('dragleave', handleDragleave);
  dropViewRef.current?.addEventListener('drop', handleDrop);
  return () => {
    dropViewRef.current?.removeEventListener('dragover', handleDragover);
    dropViewRef.current?.removeEventListener('dragleave', handleDragleave);
    dropViewRef.current?.removeEventListener('drop', handleDrop);
  };
}, [handleDragleave, handleDragover, handleDrop]);

      useMemoizedFn是ahook中的一个自定义hook，可保证函数的引用不变函数内部却能取到最新的值，从而避免频繁监听和卸载事件。不过值得注意的是，react并不推荐useMemoizedFn这种写法，他们认为该方法可能会在Concurrent Mode下有问题，不过最新的React 18 提供了一个相同功能的hook叫做useEvent

        import { useMemo, useRef } from 'react';
  type noop = (this: any, ...args: any[]) => any;

  // https://ahooks.gitee.io/zh-CN/hooks/use-memoized-fn
  function useMemoizedFn<T extends noop>(fn: T) {
    const fnRef = useRef<T>(fn);

    fnRef.current = useMemo(() => fn, [fn]);
    const memoizedFn = useRef<T>();
    if (!memoizedFn.current) {
      memoizedFn.current = function (this, ...args) {
        return fnRef.current.apply(this, args);
      } as T;
    }
    return memoizedFn.current;
  }
  export default useMemoizedFn;

    • OSS上传处理

    import { fetchOssConfig } from '@/api';
import { OssConfigType } from '@/interface';
import axios from 'axios';
import randomRangeId from './randomRangeId';

const getFileName = (fileName: string, uid: string | number = randomRangeId()) => {
  const index = fileName.lastIndexOf('.');
  // 去除文件名中的所有空格，以及特殊字符?和#和中文
  const name = fileName
    .substr(0, index)
    .replace(/\s*/g, '')
    .replace(/\?|#/g, '')
    .replace(/[\u4e00-\u9fa5]/g, '');
  const type = fileName.substr(index);
  // 兼容IE9获取文件名的问题
  if (typeof File === 'undefined') {
    return `${uid}${type}`;
  }
  return `${name}_${uid}${type}`;
};

const getOssConfig = async (configType: OssConfigType, force = false) => {
  // 设置过期时间为半小时，实际1小时
  const MAX_EXPIRES_TIME = 30 * 60 * 1000; // ms
  const ossConfigString = localStorage.getItem(`ossConfig_${configType}`);
  const ossConfig = ossConfigString ? JSON.parse(ossConfigString) : {};
  // 有缓存取缓存
  const now = Date.now();
  if (ossConfig?.expiredAt && now < ossConfig.expiredAt && !force) {
    return { ...ossConfig };
  }
  // 获取oss配置
  const userLoginId = localStorage.getItem('userLoginId') || 'default';
  const { errorCode, data, message } = await fetchOssConfig(configType, userLoginId);
  if (errorCode !== 0 || !data) throw new Error(message);
  // 更新缓存的过期时间
  data.expiredAt = Date.now() + MAX_EXPIRES_TIME;
  localStorage.setItem(`ossConfig_${configType}`, JSON.stringify(data));
  return data;
};

const beforeUpload = async (configType: OssConfigType, file: File) => {
  const { name } = file;
  const ossConfig = await getOssConfig(configType);
  const fileName = getFileName(name);
  const host = ossConfig.url.endsWith('/') ? ossConfig.url : `${ossConfig.url}/`;
  const dir = ossConfig.dir.endsWith('/') ? ossConfig.dir : `${ossConfig.dir}/`;
  const fileUrl = `${host}${dir}${fileName}`;
  const uploadOptions = {
    fileUrl,
    requestUrl: ossConfig.url as string,
    data: {
      key: `${dir}${fileName}`,
      policy: ossConfig.policy,
      OSSAccessKeyId: ossConfig.accessKeyId,
      success_action_status: 200,
      signature: ossConfig.signature,
    },
  };
  return uploadOptions;
};

const uploadImage = async (configType: OssConfigType, file: File, getProcess?: (process: number) => void) => {
  const { data, fileUrl, requestUrl } = await beforeUpload(configType, file);
  const param = new FormData();
  for (const [key, value] of Object.entries(data)) {
    param.append(key, value);
  }
  param.append('file', file);
  const config = {
    headers: { 'Content-Type': 'multipart/form-data' },
    onUploadProgress(progressEvent: any) {
      getProcess?.(Number(progressEvent.loaded / progressEvent.total));
    }
  };
  return axios.post(requestUrl, param, config).then(() => {
    return fileUrl;
  });
};

const uploadMoreImage = async (configType: OssConfigType, files: FileList | File[]) => {
  return Promise.all(Array.from(files).map((file) => uploadImage(configType, file)));
};

const transformImgToPng = (imgUrl: string) => {
  if (['png', 'jpg'].includes(imgUrl.substring(imgUrl.length - 3).toLowerCase())) return imgUrl;
  return `${imgUrl}?x-oss-process=image/format,png`;
};

export { uploadImage, uploadMoreImage, transformImgToPng };

    # UploadInput组件源码

    import React, { useEffect, useRef, useState } from 'react';
import { Toast } from 'antd-mobile';
import { transformImgToPng, uploadMoreImage } from '@/utils/uploadImage';
import classNames from 'classnames';
import { OssConfigType } from '@/interface';
import inputUploadButton from '@/assets/images/new-build-inquiry-page/input-upload-button.png';
import useMemoizedFn from '@/hooks/useMemoizedFn';
import randomRangeId from '@/utils/randomRangeId';
import Input from '../input';
import styles from './index.module.scss';

export interface IUploadInputValue {
  text?: string;
  imageUrls?: string[];
}
interface IProps extends Omit<React.ComponentProps<typeof Input>, 'onChange' | 'value'> {
  value?: IUploadInputValue;
  onChange: (value: IUploadInputValue) => void;
  inputId?: string;
  maxImageCount?: number;
  isReplaceMode?: boolean; // 替换模式只能上传一张图片
  isShowUploadButton?: boolean;
  isTransformImgToPng?: boolean;
  maxImageSize?: number; // 单位为MB
  ossType?: OssConfigType;
}
const UploadInput: React.FC<IProps> = ({
  value,
  onChange,
  maxImageCount = 1,
  isReplaceMode = false,
  isShowUploadButton = true,
  isTransformImgToPng = false,
  maxImageSize,
  inputId = randomRangeId(),
  ossType = OssConfigType.inquiry,
  ...restProps
}) => {
  const dropViewRef = useRef<HTMLInputElement>(null);
  const [isDragover, setIsDragover] = useState(false);

  const handleInputChange = (text: string) => {
    onChange({ ...value, text });
  };

  const upload = async (files?: File[]) => {
    if (!files || files?.length === 0) return;
    if (files.some((file) => !file.type.includes('image'))) {
      Toast.show(`只能上传图片`);
      return;
    }
    if (files.length > maxImageCount) {
      Toast.show(`最多只能上传${maxImageCount}张图`);
      return;
    }
    if (!isReplaceMode && files.length + (value?.imageUrls?.length || 0) > maxImageCount) {
      Toast.show(`最多只能上传${maxImageCount}张图`);
      return;
    }
    if (maxImageSize && files.some((file) => file.size > 1024 * 1024 * maxImageSize)) {
      Toast.show(`最大只能上传${maxImageSize}M的图片`);
      return;
    }
    try {
      Toast.show({
        icon: 'loading',
        content: '上传中…',
        duration: 0,
      });
      let imageUrls = await uploadMoreImage(ossType, files);
      if (isTransformImgToPng) {
        imageUrls = imageUrls.map(transformImgToPng);
      }
      Toast.clear();
      if (isReplaceMode) {
        onChange({ ...value, imageUrls });
      } else {
        onChange({ ...value, imageUrls: [...(value?.imageUrls || []), ...imageUrls] });
      }
    } catch (err) {
      Toast.clear();
      Toast.show('上传失败');
    }
  };

  // 拖拽监听
  const handleDragover = useMemoizedFn((e: DragEvent) => {
    e.preventDefault();
    if (!isDragover) {
      setIsDragover(true);
    }
  });
  const handleDragleave = useMemoizedFn(() => {
    setIsDragover(false);
  });
  const handleDrop = useMemoizedFn(async (e: DragEvent) => {
    e.preventDefault();
    e.stopPropagation();
    setIsDragover(false);
    const files = e.dataTransfer?.files && Array.from(e.dataTransfer?.files);
    upload(files);
  });
  useEffect(() => {
    dropViewRef.current?.addEventListener('dragover', handleDragover);
    dropViewRef.current?.addEventListener('dragleave', handleDragleave);
    dropViewRef.current?.addEventListener('drop', handleDrop);
    return () => {
      dropViewRef.current?.removeEventListener('dragover', handleDragover);
      dropViewRef.current?.removeEventListener('dragleave', handleDragleave);
      // eslint-disable-next-line react-hooks/exhaustive-deps
      dropViewRef.current?.removeEventListener('drop', handleDrop);
    };
  }, [handleDragleave, handleDragover, handleDrop]);

  // 粘贴监听
  const handlePaste = async (e: any) => {
    const items = e.clipboardData?.items || [];
    const files = Array.from(items)
      .map((item: any) => item.getAsFile())
      .filter((file) => file?.type?.includes('image'));
    if (files.length > 0) e.preventDefault();
    upload(files);
  };

  const handleFileInputChange = (e: any) => {
    const files = Array.from(e.target.files || []) as File[];
    e.target.value = '';
    upload(files);
  };

  const uploadButton = (
    <div
      className={styles.uploadButtonView}
      style={{ backgroundImage: `url(${inputUploadButton})` }}
    >
      <label className={styles.uploadButton} htmlFor={inputId} />
      <input
        type="file"
        multiple
        accept="image/*"
        style={{ display: 'none' }}
        id={inputId}
        onChange={handleFileInputChange}
      />
    </div>
  );
  return (
    <div className={classNames(styles.container, isDragover && styles.border)}>
      <Input
        ref={dropViewRef}
        type="text"
        onPaste={handlePaste}
        value={value?.text}
        onChange={handleInputChange}
        rightExtra={isShowUploadButton && uploadButton}
        {...restProps}
      />
    </div>
  );
};

export default UploadInput;
