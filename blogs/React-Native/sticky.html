    # 组件

    import React, {
  FC,
  memo,
  ReactNode,
  useState,
  useMemo,
  useCallback,
} from 'react';
import { Animated, LayoutChangeEvent, ViewStyle } from 'react-native';

interface IProps {
  stickyHeaderY?: number;
  stickyScrollY: Animated.Value;
  style?: ViewStyle;
  children: ReactNode;
}

const StickyView: FC<IProps> = ({
  stickyHeaderY,
  stickyScrollY = new Animated.Value(0),
  children,
  style,
}) => {
  const [stickyLayoutY, setStickyLayoutY] = useState(0);

  const translateY = useMemo(() => {
    const y = stickyHeaderY ?? stickyLayoutY;
    const translateY = stickyScrollY.interpolate({
      inputRange: [-1, 0, y, y + 1],
      outputRange: [0, 0, 0, 1],
    });
    return translateY;
  }, [stickyHeaderY, stickyLayoutY]);

  const onLayout = useCallback((event: LayoutChangeEvent) => {
    setStickyLayoutY(event.nativeEvent.layout.y);
  }, []);

  return (
    <Animated.View
      onLayout={onLayout}
      style={[style, { zIndex: 1000, transform: [{ translateY }] }]}>
      {children}
    </Animated.View>
  );
};

export default memo(StickyView);

    import PageWrapper from '@components/common/page-wrapper/PageWrapper';
import StickyView from '@components/common/sticky-view/StickyView';
import { ICommonProps } from '@typings/common';
import { getRealSize as S } from '@utils/utility';
import React, { useRef } from 'react';
import { View, StyleSheet, Text, Animated } from 'react-native';

const Demo = ({ navigation }: ICommonProps) => {
  const stickyScrollYRef = useRef(new Animated.Value(0));
  return (
    <PageWrapper title="测试" navigation={navigation}>
      <Animated.ScrollView
        stickyHeaderIndices
        onScroll={
          Animated.event(
            [
              {
                nativeEvent: { contentOffset: { y: stickyScrollYRef.current } }, // 记录滑动距离
              },
            ],
            { useNativeDriver: true },
          ) // 使用原生动画驱动
        }>
        <View style={styles.header}></View>
        <StickyView
          style={styles.container}
          stickyScrollY={stickyScrollYRef.current}>
          <Text>666</Text>
        </StickyView>
