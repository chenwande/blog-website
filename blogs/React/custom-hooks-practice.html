    Custom Hooks 实践

    槐序 2022/3/7 reactreact-hook

    自定义 Hook 是 React 中可以使用其他 Hook 的一种函数，通过它可以将组件逻辑提取到可重用的函数中。

    注意点

    • 自定义 Hook 必须以“use”开头
    • 两个组件中使用相同的 Hook 不会共享 state
    • 自定义 Hook 和其他 React Hook 一样必须要在函数组件的顶层使用。

    # React 生命周期相关的自定义 Hook 实现

    • useMount

      只在组件初始化时执行的 Hook。模拟 componentDidMount 生命周期

    • useUpdate

      只在组件更新时执行的 Hook，挂载时不执行。模拟 componentDidUpdate 生命周期

    • useUnmount

      在组件卸载（unmount）时执行的 Hook。模拟 componentWillUnmount 生命周期

    • useMountedRef

      返回一个组件是否挂在的 ref。

    • useForceUpdate

      返回一个可触发页面重新渲染的函数。模拟this.forceUpdate()函数

    import { useReducer, useEffect, useRef } from 'react';
/** 模拟 componentDidMount 生命周期 */
export const useMount = (fn: () => void) => {
  useEffect(() => {
    fn();
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, []);
};

/** 模拟 componentDidUpdate 生命周期 */
export const useUpdate = (fn: () => void) => {
  const isMounted = useRef(false);
  useEffect(() => {
    if (!isMounted.current) {
      isMounted.current = true;
    } else {
      fn();
    }
  });
};

/** 模拟 componentWillUnmount 生命周期 */
export const useUnmount = (fn: () => void) => {
  // eslint-disable-next-line react-hooks/exhaustive-deps
  useEffect(() => fn, []);
};

/**
 * 用于判断组件是否已经挂载
 */
export const useMountedRef = () => {
  const ref = useRef(false);
  useEffect(() => {
    ref.current = true;
    return () => {
      ref.current = false;
    };
  }, []);
  return ref;
};

/**
 * 模拟 forceUpdate 方法, 强制更新组件
 * @returns 强制更新函数
 */
export const useForceUpdate = () => {
  // 1、
  const [_ignore, forceUpdate] = useReducer((s: number) => s + 1, 0);

  // 2、
  // const [, set] = useState({});
  // const forceUpdate = () => set({});

  // 3、
  // const [, set] = useState(false);
  // const forceUpdate = () => set(bool => !boole);

  return forceUpdate;
};

export default {
  useMount,
  useUpdate,
  useUnmount,
  useMountedRef,
  useForceUpdate,
};

    # 一些自定义 hook 的列子

    # useDebounce

    • 功能：传入一个value值和 number 类型的delay，返回一个 delay 毫秒后发生变化的debouncedValue，当delay毫秒内value发生变化，则重新计时。
    • 用途：不带按钮的搜索功能
    import { useEffect, useState } from 'react';

const useDebounceValue = <V>(value: V, delay: number = 500) => {
  const [debouncedValue, setDebouncedValue] = useState(value);

  useEffect(() => {
    // 每次在value变化以后，设置一个定时器
    const timeout = setTimeout(() => setDebouncedValue(value), delay);
    // 每次在上一个useEffect处理完以后再运行
    return () => clearTimeout(timeout);
  }, [value, delay]);

  return debouncedValue;
};

export default useDebounceValue;

    # useFormData

    • 功能：统一管理表单数据
    • 参数：接受一个初始的表单数据对象initData，返回当前的表单对象数据formData和setFormItem、resetFormItem两个处理函数。setFormItem接受一个key和value参数，resetFormItem不传参数。
     import { useRef, useState, useMemo } from 'react';

 // 定义
 export const useFormData = (initData: { [key: string]: string } = {}) => {
   const formData = useRef(initData);
   const [, update] = useState(false);

   const handleForm = useMemo(() => {
     const setFormItem = (key: string, value: any) => {
       formData.current[key] = value;
       update(v => !v);
     };
     const resetForm = () => {
       Object.keys(formData.current).forEach(key => (formData.current[key] = ''));
       update(v => !v);
     };
     return [setFormItem, resetForm] as const;
   }, []);
   return [formData.current, ...handleForm] as const;
 };

    # useAsync

    • 功能：用于处理异步调用
    • 参数：无入参，返回Error类型的error和boolean类型的isLoading以及data和run函数，run函数接受一个Promise作为参数，使用run运行Promise后，获取error、isLoading和data的值。
    import { useCallback, useEffect, useRef, useState } from 'react';

export const useAsync = <T>() => {
  const [isLoading, setIsLoading] = useSafeState(false);
  const [error, setError] = useSafeState<Error | null>(null);
  const [data, setData] = useSafeState<T | null>(null);
  const retryRef = useRef<null | (() => Promise<T>)>(null);

  const run = useCallback(
    (promise: Promise<T>, runConfig?: { retry: () => Promise<T> }) => {
      retryRef.current = runConfig?.retry || null;
      setIsLoading(true);
      setError(null);
      promise
        .then((res) => {
          setData(res);
        })
        .catch((err) => {
          setError(err);
        })
        .finally(() => {
          setIsLoading(false);
        });
    },
    [setData, setError, setIsLoading],
  );

  const retry = useCallback(() => {
    if (!retryRef.current) throw new Error('不存在可用的retry');
    run(retryRef.current(), { retry: retryRef.current });
  }, [run]);

  return {
    isLoading,
    error,
    data,
    run,
    retry,
  };
};

const useMountedRef = () => {
  const mountedRef = useRef(false);
  useEffect(() => {
    if (!mountedRef.current) {
      mountedRef.current = true;
    }
    return () => {
      mountedRef.current = false;
    };
  }, []);
  return mountedRef;
};

const useSafeState = <S>(initialState: S | (() => S)) => {
  const [state, set] = useState(initialState);
  const mountedRef = useMountedRef();
  return [
    state,
    useCallback((...params: Parameters<typeof set>) => {
      if (mountedRef.current) {
        set(...params);
      }
    }, [mountedRef]),
  ] as const;
};

    # useCountdown

    • 功能：一个用于管理倒计时的 Hook。
    import dayjs from 'dayjs';
import { useEffect, useMemo, useState } from 'react';

export type TDate = Date | number | string | undefined;

export type Options = {
  targetDate: TDate;
  interval?: number;
  onEnd?: () => void;
};

export interface FormattedRes {
  days: number;
  hours: number;
  minutes: number;
  seconds: number;
  milliseconds: number;
}

const calcLeft = (t?: TDate) => {
  if (!t) {
    return 0;
  }
  // https://stackoverflow.com/questions/4310953/invalid-date-in-safari
  const left = dayjs(t).valueOf() - new Date().getTime();
  if (left < 0) {
    return 0;
  }
  return left;
};

const parseMs = (milliseconds: number): FormattedRes => {
  return {
    days: Math.floor(milliseconds / 86400000),
    hours: Math.floor(milliseconds / 3600000) % 24,
    minutes: Math.floor(milliseconds / 60000) % 60,
    seconds: Math.floor(milliseconds / 1000) % 60,
    milliseconds: Math.floor(milliseconds) % 1000,
  };
};

const useCountdown = (options?: Options) => {
  const { targetDate, interval = 1000, onEnd } = options || {};

  const [timeLeft, setTimeLeft] = useState(() => calcLeft(targetDate));

  useEffect(() => {
    if (!targetDate) {
      // for stop
      setTimeLeft(0);
      return;
    }

    // 立即执行一次
    setTimeLeft(calcLeft(targetDate));

    const timer = setInterval(() => {
      const targetLeft = calcLeft(targetDate);
      setTimeLeft(targetLeft);
      if (targetLeft === 0) {
        clearInterval(timer);
        onEnd?.();
      }
    }, interval);

    return () => clearInterval(timer);
  }, [targetDate, interval]);

  const formattedRes = useMemo(() => {
    return parseMs(timeLeft);
  }, [timeLeft]);

  return [timeLeft, formattedRes] as const;
};

export default useCountdown;
