    Custom Hooks

    槐序 2021/11/12 reactreact-hook

    自定义 Hook是React中可以使用其他Hook的一种函数，通过它可以将组件逻辑提取到可重用的函数中。

    注意点

    • 自定义Hook必须以“use”开头
    • 两个组件中使用相同的Hook不会共享state
    • 自定义Hook和其他React Hook一样必须要在函数组件的顶层使用。

    # React生命周期相关的自定义Hook实现

    • useMount

    • useUpdate

    • useMountAndUpdate

    • useUnmount

    • useIsMounted

    • useForceUpdate

     import { useReducer, useEffect, useRef } from 'react';
 
 /** 模拟 componentDidMount 生命周期 */
 export const useMount = (fn: () => void) => {
   useEffect(() => {
     fn();
     // eslint-disable-next-line react-hooks/exhaustive-deps
   }, []);
 };
 
 /** 模拟 componentDidUpdate 生命周期 */
 export const useUpdate = (fn: () => void) => {
   const isMounted = useRef(false);
   useEffect(() => {
     if (!isMounted.current) {
       isMounted.current = true;
     } else {
       fn();
     }
   });
 };
 
 /** 模拟 componentDidMount和componentDidUpdate 生命周期 */
 export const useMountAndUpdate = (fn: () => void) => {
   useEffect(() => {
     fn();
     // eslint-disable-next-line react-hooks/exhaustive-deps
   },);
 };
 
 /** 模拟 componentWillUnmount 生命周期 */
 export const useUnmount = (fn: () => void) => {
   // eslint-disable-next-line react-hooks/exhaustive-deps
   useEffect(() => fn, []);
 };
 
 /**
  * 用于判断组件是否已经挂载
  * @returns fn: 调用获取是否挂载布尔值
  */
 export const useIsMounted = () => {
   const ref = useRef(false);
   useEffect(() => {
     ref.current = true;
     return () => {
       ref.current = false;
     };
   }, []);
   return () => ref.current;
 };
 
 /**
  * 模拟 forceUpdate 方法, 强制更新组件
  * @returns 强制更新函数
  */
 export const useForceUpdate = () => {
   const [, forceUpdate] = useReducer((s: number) => s + 1, 0);
 
   return forceUpdate;
 };

    # 一些自定义hook的列子

    # useDebounce

    • 功能：传入一个value值和number类型的delay，返回一个delay毫秒后发生变化的debouncedValue，当delay毫秒内value发生变化，则重新计时。
    • 用途：不带按钮的搜索功能
     // 闭包实现
 const debounce = (func, delay) => {
    let timeout;
    return (...param) => {
      if (timeout) {
        clearTimeout(timeout);
      }
      timeout = setTimeout(() => func(...param), delay);
    };
  };
 
 // hook实现
 export const useDebounce = <T>(value: T, delay?: number) => {
   const [debouncedValue, setDebouncedValue] = useState(value);
   useEffect(() => {
     // 每次在value变化以后，设置一个定时器
     const timeout = setTimeout(() => setDebouncedValue(value), delay);
     // 每次在上一个useEffect处理完以后再运行
     return () => clearTimeout(timeout);
   }, [delay, value]);
   return debouncedValue;
 };
         
 // 使用
 export default function Demo() {
   const [keyword, setKeyeord] = useState("");
   const debouncedKeyword = useDebounce(keyword, 500);
   useEffect(() => {
     const fetchData = (keyword) => {
       console.log(keyword);
       // ...调用接口
     };
     fetchData(debouncedKeyword);
   }, [debouncedKeyword]);
   return (
     <div>
       <input
         value={keyword}
         onChange={(event) => setKeyeord(event.target.value)}
       />
     </div>
   );
 }

    # useUrlParam

    • 功能：web页面获取和修改url的search参数，该hook返回当前的url参数对象和一个修改函数（替换修改）
    • 该hook依赖于react-router库
     import { cleanObject } from './../functions/cleanObject';
 import qs from 'qs';
 import { useEffect, useState } from 'react';
 import { useHistory, useLocation } from 'react-router';
 
 // 定义
 interface UrlState {
   [key: string]: any;
 }
 const useUrlState = (
   param: { [key: string]: any } = Object.fromEntries(new URLSearchParams(location.search))
 ) => {
   const history = useHistory();
   const location = useLocation();
   const [urlState, setUrlState] = useState(cleanObject(param));
   useEffect(() => {
     history.replace({
       hash: location.hash,
       search: qs.stringify(cleanObject(urlState)) || '?',
     });
   }, [urlState]);
   return [urlState, setUrlState] as const;
 };
 export default useUrlState; 
 
 // 使用
 export default function Demo = () => {
   const [param, setUrlParam] = useUrlState();
   const users = [
     {
       id: 1,
       name: '高修文',
     },
     {
       id: 2,
       name: '熊天成',
     },
     {
       id: 3,
       name: '郑华',
     },
     {
       id: 4,
       name: '王文静',
     },
   ];
   return (
     <form>
       <input
         type="text"
         value={param.name || ''}
         onChange={e => setUrlParam({ ...param, name: e.target.value })}
       />
       <select
         value={param.personId || ''}
         onChange={e => setUrlParam({ ...param, personId: e.target.value })}
       >
         <option value="">负责人</option>
         {users.map(user => (
           <option value={user.id} key={user.id}>
             {user.name}
           </option>
         ))}
       </select>
     </form>
   );
 };

    # useFormData

    • 功能：统一管理表单数据
    • 参数：接受一个初始的表单数据对象initData，返回当前的表单对象数据formData和setFormItem、resetFormItem两个处理函数。setFormItem接受一个key和value参数，resetFormItem不传参数。
     import { useRef, useState, useMemo } from 'react';
 
 // 定义
 export const useFormData = (initData: { [key: string]: string } = {}) => {
   const formData = useRef(initData);
   const [, update] = useState(false);
 
   const handleForm = useMemo(() => {
     const setFormItem = (key: string, value: any) => {
       formData.current[key] = value;
       update(v => !v);
     };
     const resetForm = () => {
       Object.keys(formData.current).forEach(key => (formData.current[key] = ''));
       update(v => !v);
     };
     return [setFormItem, resetForm] as const;
   }, []);
   return [formData.current, ...handleForm] as const;
 };
 
 // 使用
 export default function Demo() {
   const [formData, setFormItem, resetForm] = useFormData({
     title1: "",
     title2: "",
     title3: ""
   });
   const { title1, title2, title3 } = formData;
   return (
     <div>
       <div>
         <label htmlFor="title1">标题1</label>
         <input
           type="text"
           id="title1"
           value={title1}
           onChange={(e) => setFormItem("title1", e.target.value)}
         />
       </div>
       <div>
         <label htmlFor="title2">标题2</label>
         <input
           type="text"
           id="title2"
           value={title2}
           onChange={(e) => setFormItem("title2", e.target.value)}
         />
       </div>
       <div>
         <label htmlFor="title3">标题3</label>
         <input
           type="text"
           id="title3"
           value={title3}
           onChange={(e) => setFormItem("title3", e.target.value)}
         />
       </div>
       <button type="button" onClick={(e) => console.log(formData)}>
         打印
       </button>
       <button type="button" onClick={resetForm}>
         重置
       </button>
     </div>
   );
 }

    # useAsync

    • 功能：用于处理异步调用
    • 参数：无入参，返回Error类型的error和boolean类型的loading以及data和run函数，run函数接受一个Promise作为参数，使用run运行Promise后，获取error、loading和data的值。
     import { useCallback, useState } from 'react';
 
 // 定义
 export const useAsync = () => {
   const [data, setData] = useState<any>(null);
   const [error, setError] = useState<Error | null>(null);
   const [loading, setLoading] = useState(false);
   const run = useCallback((promise: Promise<any>) => {
     setLoading(true);
     promise
       .then(data => {
         setData(data);
         setLoading(false);
       })
       .catch(e => {
         setError(e);
         setLoading(false)
       });
   }, []);
   return {
     run,
     data,
     error,
     loading,
   };
 };
 
 // 使用
 export const Demo = () => {
   const { run: fetchData, data, loading: fetchLoading, error: fetchError } = useAsync();
   const { run, loading: postLoading, error: postError } = useAsync();
   useEffect(() => {
     fetchData(http('users/2'));
   }, []);
   return (
     <div>
       <div>
         <button
           type="button"
           onClick={e => {
             run(
               http('users', {
                 method: 'POST',
                 data: {
                   // name: '姓名',
                 },
               })
             );
           }}
         >
           {postLoading ? 'loading...' : '添加'}
         </button>
       </div>
       {fetchLoading ? (
         'loading...'
       ) : fetchError ? (
         <p style={{ color: 'red' }}>{fetchError.message}</p>
       ) : (
         <div>{JSON.stringify(data)}</div>
       )}
     </div>
   );
 };
